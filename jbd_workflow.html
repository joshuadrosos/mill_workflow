 filter(START >= "1990-12-31 18:00:00" & 
          START <= "2040-01-01 00:00:00") %>%
 filter(STOP >= "1990-12-31 18:00:00" & 
          STOP <= "2040-01-01 00:00:00") %>%
 filter(ENCOUNTERCLASS %in% c("emergency", "inpatient", "outpatient", "urgentcare"))
